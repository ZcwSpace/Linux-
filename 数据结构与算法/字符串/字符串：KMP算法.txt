#include <stdio.h>
int kmp(char s[], char t[]);
void get_next(char t[], int next[]);
int main()
{
	char s[20], t[10];
	printf("请输入主串内容：");
	scanf_s("%s", s, 20);
	printf("请输入子串内容：");
	scanf_s("%s", t, 10);
	printf("%d", kmp(s, t));
	return 0;
}
int kmp(char s[], char t[])
{
	int next[10], i = 0, j = 0;
	get_next(t, next);
	while (s[i] != '\0' && t[j] != '\0')
	{
		if (j == -1 || s[i] == t[j])
		{
			i++;
			j++;
		}
		else
		{
			j = next[j];
		}
	}
	if (t[j] == '\0')
	{
		return 1;
	}
	else
	{
		return 0;
	}
}
void get_next(char t[], int next[])
{
	int i = 0, j = -1;
	next[0] = -1;
	while (t[i] != '\0')
	{
		if (j == -1 || t[i] == t[j])
		{
			i++;
			j++;
			if (t[i] != t[j])
			{
				next[i] = j;
			}
			else
			{
				next[i] = next[j];
			}
		}
		else
		{
			j = next[j];
		}
	}
	for (i = 0;t[i] != '\0';i++)
	{
		printf("%d ", next[i]);
	}
	printf("\n");
}
