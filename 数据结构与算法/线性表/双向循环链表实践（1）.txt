#include <stdio.h>
#include <stdlib.h>
struct b
{
	char character;
	struct b *prior;
	struct b *next;
};
#define LENGTH sizeof(struct b)
struct b *create();
struct b *change(struct b *L, int n);
int main()
{
	struct b *num, *head;
	int n;
	num = create();
	printf("请输入一个数：");
	scanf_s("%d", &n);
	num = change(num, n);
	head = num->next;
	num = num->next;
	do
	{
		printf("%c ", num->character);
		num = num->next;
	} while (num != head);
	return 0;
}
struct b *create()
{
	struct b *p, *p1, *head;
	head = NULL;
	int n = 0;
	char i='A';
	p = p1 = (struct b *)malloc(LENGTH);
	p->character = i;
	i++;
	while (p->character<='Z')
	{
		n++;
		if (n == 1)
		{
			head = p;
		}
		else
		{
			p1->next = p;
			p->prior = p1;
			p1 = p;
		}
		p = (struct b *)malloc(LENGTH);
		p->character = i;
		i++;
	}
	p1->next = head;
	head->prior = p1;
	return p1;
}
struct b *change(struct b *L, int n)
{
	int i;
	struct b *p;
	p = L;
	if (n > 0)
	{
		for (i = 1;i <= n;i++)
		{
			p = p->next;
		}
		L = p;
	}
	else if (n < 0)
	{
		for (i = -1;i >= n;i--)
		{
			p = p->prior;
		}
		L = p;
	}
	return L;
}
