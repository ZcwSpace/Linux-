#include<stdio.h>
int judge(int row, int line, int(*a)[4]);
void n(int row, int line, int(*a)[4]);
int count = 0;
int main()
{
	int a[4][4];
	int j;
	for (j = 0;j < 4;j++)
	{
		a[0][j] = j + 1;
	}
	n(1, 0, a);
}
void n(int row, int line, int(*a)[4])
{
	int a1[4][4], i, j, k;
	for (i = 0;i < 4;i++)
	{
		for (j = 0;j < 4;j++)
		{
			a1[i][j] = a[i][j];
		}
	}
	if (row == 4 && line == 0)
	{
		printf("第%d次\n", ++count);
		for (i = 0;i < 4;i++)
		{
			for (j = 0;j < 4;j++)
			{
				printf("%d ", a1[i][j]);
			}
			printf("\n");
		}
		printf("\n");
	}
	else
	{
		for (k = 1;k <= 4;k++)
		{
			a1[row][line] = k;
			if (judge(row, line, a1))
			{
				if (line == 3)
				{
					n(row + 1, 0, a1);
				}
				else
				{
					n(row, line + 1, a1);
				}
			}
		}
	}
}
int judge(int row, int line, int(*a)[4])
{
	int i, j;
	for (j = 0;j < line;j++)/*左*/
	{
		if (a[row][j] == a[row][line])
		{
			return 0;
		}
	}
	for (i = 0;i < row;i++)/*上*/
	{
		if (a[i][line] == a[row][line])
		{
			return 0;
		}
	}
	if (row == line)
	{
		for (i = 0, j = 0;i < row && j < line;i++, j++)/*左上*/
		{
			if (a[i][j] == a[row][line])
			{
				return 0;
			}
		}
	}
	if (row == 1 && line == 2 || row == 2 && line == 1 || row == 3 && line == 0)
	{
		for (i = row - 1, j = line + 1;i >= 0 && j <= 3;i--, j++)
		{
			if (a[i][j] == a[row][line])
			{
				return 0;
			}
		}
	}
	return 1;
}
